create table [user]
(
    firstname varchar(50),
    lastname  varchar(50),
    email     varchar(50),
    mobile    varchar(15),
    id        uniqueidentifier not null
        constraint user_pk
            primary key
)
go

create unique index user_id_uindex
    on [user] (id)
go

create unique index user_mobile_uindex
    on [user] (mobile)
go

create unique index user_email_uindex
    on [user] (email)
go
