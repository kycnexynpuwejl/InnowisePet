create table [order]
(
    firstname varchar(50)      not null,
    lastname  varchar(50)      not null,
    address   varchar(50)      not null,
    city      varchar(50)      not null,
    country   varchar(50)      not null,
    id        uniqueidentifier not null
        constraint order_pk
            primary key
)
go

create unique index order_id_uindex
    on [order] (id)
go
